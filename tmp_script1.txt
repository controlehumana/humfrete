_auth.onAuthStateChanged(async (user) => {
  if (user) {
    showLoading();

      const uDoc = await _db.collection('users').doc(user.uid).get();

async function _loadData(empresas) {
  if (!empresas.length) throw new Error('Nenhuma empresa autorizada para este usuario.');
  const datas = await Promise.all(empresas.map(async (emp) => {
    const url = await _st.ref('data/' + emp + '.json').getDownloadURL();
    const r = await fetch(url);
    if (!r.ok) throw new Error('Falha ao baixar ' + emp + '.json');
    return r.json();
  }));
  DATA = _mergeData(datas);
}

function _mergeData(datas) {
  const totFrete = datas.reduce((s,d)=>s+(d.resumo?.valor_total_frete||0),0);
  const totNfe   = datas.reduce((s,d)=>s+(d.resumo?.nfe_com_cte||0),0);
  return {
    gerado_em: datas[0]?.gerado_em || '',
    cnpj_map: Object.assign({}, ...datas.map(d=>d.cnpj_map||{})),
    por_nat_op: datas[0]?.por_nat_op || [],
    detalhes: datas.flatMap(d=>d.detalhes||[]),
    ctes_nao_vinculados: datas.flatMap(d=>d.ctes_nao_vinculados||[]),
    cancelados_data: datas.flatMap(d=>d.cancelados_data||[]),
    cte_cancelados_chaves: [...new Set(datas.flatMap(d=>d.cte_cancelados_chaves||[]))],
    compras: datas.flatMap(d=>d.compras||[]),
    devolucoes_mkt: datas.flatMap(d=>d.devolucoes_mkt||[]),
    transf_sem_cte: datas.flatMap(d=>d.transf_sem_cte||[]),
    transf_fat: Object.assign({}, ...datas.map(d=>d.transf_fat||{})),
    resumo: {
      ...(datas[0]?.resumo||{}),
      valor_total_frete: Math.round(totFrete*100)/100,
      total_faturamento: Math.round(datas.reduce((s,d)=>s+(d.resumo?.total_faturamento||0),0)*100)/100,
      nfe_com_cte: totNfe,
      media_frete: totNfe ? Math.round(totFrete/totNfe*100)/100 : 0,
    }
  };
}

function _applyPermissions() {
  if (!_currentUser) return;
  const adminBtn = document.getElementById('tab-btn-admin');
  if (adminBtn) adminBtn.style.display = _currentUser.isAdmin ? '' : 'none';
  if (_currentUser.isAdmin) return;
  const tabs = _currentUser.tabs;
  if (tabs && Array.isArray(tabs)) {
    document.querySelectorAll('.tab-btn[data-tab]').forEach(btn => {
      const t = btn.dataset.tab;
      if (t !== 'admin' && !tabs.includes(t)) btn.style.display = 'none';
    });
    const first = document.querySelector('.tab-btn[data-tab]:not([style*="none"])');
    if (first && !first.classList.contains('active')) {
      document.querySelectorAll('.tab-btn.active,.tab-panel.active').forEach(el=>el.classList.remove('active'));
      first.classList.add('active');
      const panel = document.getElementById('tab-'+first.dataset.tab);
      if (panel) panel.classList.add('active');
    }
  }
}

function showLoginScreen(err) {
  document.getElementById('login-overlay').style.display = 'flex';
  document.getElementById('app').style.display = 'none';
  document.getElementById('loading-overlay').style.display = 'none';
  const el = document.getElementById('login-error');
  if (err) { el.textContent = err; el.style.display = 'block'; }
  else { el.style.display = 'none'; }
}
function showLoading() {
  document.getElementById('login-overlay').style.display = 'none';
  document.getElementById('loading-overlay').style.display = 'flex';
}
function hideLoading() {
  document.getElementById('loading-overlay').style.display = 'none';
}
async function doLogin(e) {
  e.preventDefault();
  const email = document.getElementById('login-email').value;
  const pass  = document.getElementById('login-pass').value;
  const btn   = document.getElementById('login-btn');
  const err   = document.getElementById('login-error');
  btn.disabled = true; btn.textContent = 'Entrando...';
  err.style.display = 'none';
  try {
    await _auth.signInWithEmailAndPassword(email, pass);
  } catch(e2) {
    err.textContent = e2.code === 'auth/wrong-password' || e2.code === 'auth/user-not-found'
      ? 'E-mail ou senha incorretos.' : e2.message;
    err.style.display = 'block';
    btn.disabled = false; btn.textContent = 'Entrar';
  }
}
async function doForgotPassword(e) {
  e.preventDefault();
  const email = document.getElementById('login-email').value;
  if (!email) { alert('Digite o e-mail primeiro.'); return; }
  try {
    await _auth.sendPasswordResetEmail(email);
    alert('E-mail de recuperacao enviado para ' + email + '. Verifique sua caixa de entrada.');
  } catch(e2) { alert('Erro: ' + e2.message); }
}
async function doLogout() {
  if (confirm('Deseja sair do sistema?')) await _auth.signOut();
}
